OptionMenu "WeaponTiltOptions" 
{
	Title "$UWT_TITL"
    StaticText "Universal Lean, Weapon Lowering, Wall-Detection & Rotational-Tilting", gold
   
    StaticText ""

	Slider "$UWT_VELOCITY", "wt_rollvelocity", 0.50, 2.0, 0.01, 2

	Slider "$UWT_RESISTANCE", "wt_rollresistance", 0.10, 0.25, 0.01, 2

    StaticText ""

    Slider "$UWT_LOWERING_INTENSITY", "wt_lowering_intensity", 0.0, 2.0, 0.05, 2
    StaticText "0 = No lowering, 1 = Default, 2 = Double", "Teal"
    StaticText "Controls how much the weapon lowers when strafing.", "Teal"

    Slider "$UWT_WALLLOWERING", "wt_walllowering", 0.0, 16.0, 0.5, 1
    StaticText "Higher values make the weapon dip more.", "Teal"
    StaticText "How much extra the weapon lowers when facing a wall.", "Teal"
    StaticText ""

    Option "$UWT_CAP", "wt_cap", "OnOff", white

    Slider "$UWT_CAPAMOUNT", "wt_rollcap", 1, 5, 0.5, 2

    StaticText ""
    
    Option "$UWT_DEBUG", "wt_debug", "OnOff"
    Option "$UWT_OFFSET", "wt_offset", "YesNo"

}

AddOptionMenu "OptionsMenu"
{
	Submenu "$UWT_TITL", "WeaponTiltOptions"
	StaticText ""
}
